Estación de Gasolina

const clave = prompt("Ingrese la clave de la gasolina (1, 2 o 3):");
const litros = parseFloat(prompt("Ingrese los litros consumidos:"));
let precio = 0;

switch (clave) {
    case "1": precio = 45; break;
    case "2": precio = 47; break;
    case "3": precio = 48.5; break;
    default: alert("Clave inválida."); break;
}

if (precio > 0) {
    const total = litros * precio;
    alert("Total a pagar: Q" + total.toFixed(2));
}

Mes y Días

const mes = parseInt(prompt("Ingrese el número de mes (1-12):"));
const anio = parseInt(prompt("Ingrese el año:"));
const nombresMeses = ["enero", "febrero", "marzo", "abril", "mayo", "junio", 
                      "julio", "agosto", "septiembre", "octubre", "noviembre", "diciembre"];
let dias;

if (mes === 2) {
    dias = (anio % 4 === 0 && (anio % 100 !== 0 || anio % 400 === 0)) ? 29 : 28;
} else if ([4, 6, 9, 11].includes(mes)) {
    dias = 30;
} else {
    dias = 31;
}

alert(nombresMeses[mes - 1] + " tiene " + dias + " días.");

Ciudadanos y Votación

let huehue = [], quetzal = [], guate = [];
let input;

do {
    let dept = prompt("Ingrese departamento (Huehuetenango, Quetzaltenango, Guatemala o SALIR):").toUpperCase();
    if (dept === "SALIR") break;

    if (["HUEHUETENANGO", "QUETZALTENANGO", "GUATEMALA"].includes(dept)) {
        let edad = parseInt(prompt("Ingrese la edad:"));
        if (edad >= 18) alert("Puede votar");
        else alert("No puede votar");

        if (dept === "HUEHUETENANGO") huehue.push(edad);
        else if (dept === "QUETZALTENANGO") quetzal.push(edad);
        else if (dept === "GUATEMALA") guate.push(edad);
    } else {
        alert("Departamento inválido.");
    }
} while (true);

// Mostrar resumen
function resumen(depto, arr) {
    let mayores = arr.filter(e => e >= 18).length;
    let menores = arr.filter(e => e < 18).length;
    let prom = arr.reduce((a,b) => a+b, 0) / (arr.length || 1);
    console.log(depto + ": Total=" + arr.length + ", Mayores=" + mayores + ", Menores=" + menores + ", Promedio=" + prom.toFixed(2));
}
resumen("Huehuetenango", huehue);
resumen("Quetzaltenango", quetzal);
resumen("Guatemala", guate);

Índice de Masa Corporal

let peso = parseFloat(prompt("Ingrese su peso en kilogramos:"));
let altura = parseFloat(prompt("Ingrese su altura en metros:"));
let imc = peso / (altura * altura);
let estado = "";

if (imc < 18.5) estado = "Bajo peso";
else if (imc < 24.9) estado = "Normal";
else if (imc < 29.9) estado = "Sobrepeso";
else estado = "Obesidad";

alert("Su IMC es: " + imc.toFixed(2) + " - " + estado);
